    function Toast({ toast, onClose }) {
      useEffect(() => {
        if (!toast) return;
        const id = setTimeout(onClose, toast.type === 'error' ? 4500 : 2500);

      }, [toast, onClose]);

      const isErr = toast.type === 'error';
      return (
        <div className="fixed top-4 left-1/2 -translate-x-1/2 z-[200] px-2 max-w-md w-full pointer-events-none">
          <div className={`pointer-events-auto rounded-lg px-4 py-3 shadow-2xl backdrop-blur flex items-start gap-3 border ${isErr ? 'bg-red-950/95 border-red-500/40 text-red-100' : 'bg-emerald-950/95 border-emerald-500/40 text-emerald-100'}`} style={{ animation: 'slideDown 0.2s ease-out' }}>

    // ===== MAIN APP =====
    function App() {
      const [view, setView] = useState('home');
      const [currentPlayer, setCurrentPlayer] = useState(null);
      const [isAdmin, setIsAdmin] = useState(false);

      const [config, setConfig] = useState({
        adminPassword: '',
        deadline: DEFAULT_DEADLINE,
        scoring: DEFAULT_SCORING,
        initialized: false,
      });
      const [players, setPlayers] = useState([]);
      const [results, setResults] = useState({});
      const [loading, setLoading] = useState(true);
      const [savingState, setSavingState] = useState('');
      const [toast, setToast] = useState(null);
      const [confirmDlg, setConfirmDlg] = useState(null);

      // We keep all state in one bin record
      const stateRef = useRef({ config, players, results });

      const showToast = useCallback((message, type = 'success') => setToast({ message, type }), []);
      const askConfirm = useCallback((title, message, onConfirm, danger = false) => setConfirmDlg({ title, message, onConfirm, danger }), []);

      const loadState = useCallback(async () => {
        const data = await loadFromBin();
        if (data && typeof data === 'object') {
          if (data.config) {
            setConfig(prev => ({ ...prev, ...data.config }));
            stateRef.current.config = data.config;
          }
          if (Array.isArray(data.players)) {
            setPlayers(data.players);
            stateRef.current.players = data.players;
          }
          if (data.results && typeof data.results === 'object') {
            setResults(data.results);
            stateRef.current.results = data.results;
          }
        }
        setLoading(false);
      }, []);

      useEffect(() => {
        loadState();
        const interval = setInterval(loadState, 30000);
        return () => clearInterval(interval);
      }, [loadState]);

      // Salvează tot bin-ul (config + players + results)
      const saveAll = async (partial) => {
        const newState = {
          config: partial.config ?? stateRef.current.config,
          players: partial.players ?? stateRef.current.players,
          results: partial.results ?? stateRef.current.results,
        };
        stateRef.current = newState;
        setSavingState('saving');
        const ok = await saveToBin(newState);
        setSavingState(ok ? 'saved' : '');
        if (!ok) showToast('Eroare la salvare. Verifică conexiunea și reîncearcă.', 'error');
        setTimeout(() => setSavingState(''), 1500);
        return ok;
      };

      const saveConfig = async (newCfg) => {
        setConfig(newCfg);
        return saveAll({ config: newCfg });
      };
      const savePlayers = async (newPlayers) => {
        setPlayers(newPlayers);
        return saveAll({ players: newPlayers });
      };
      const saveResults = async (newResults) => {
        setResults(newResults);
        return saveAll({ results: newResults });
      };

      const isLocked = useMemo(() => {
        if (!config.deadline) return false;
        const d = new Date(config.deadline);
        if (isNaN(d.getTime())) return false;
        return new Date() >= d;
      }, [config.deadline]);

      const standings = useMemo(() => {
        const arr = players.map(p => {
          let total = 0, exact = 0, diff = 0, outcome = 0, wrong = 0, played = 0;
          for (const m of MATCHES) {
            const pred = p.predictions?.[m.n];
            const real = results[m.n];
            const pts = calcPoints(pred, real, config.scoring);
            if (pts == null) continue;
            played++; total += pts;
            if (pts === config.scoring.exact) exact++;
            else if (pts === config.scoring.diff) diff++;
            else if (pts === config.scoring.outcome) outcome++;
            else wrong++;
          }
          return { name: p.name, total, exact, diff, outcome, wrong, played };
        });
        return arr.sort((a, b) => b.total - a.total || b.exact - a.exact || b.diff - a.diff);
      }, [players, results, config.scoring]);

      if (!isStorageConfigured()) return <ConfigErrorScreen />;

      if (loading) {
        return (
          <div className="min-h-screen flex items-center justify-center">
            <div className="text-stone-400 font-mono text-sm tracking-widest animate-pulse">SE ÎNCARCĂ...</div>
          </div>
        );
      }

      if (!config.initialized) {
        return <>
          <SetupView onSave={async (cfg) => {
            await saveConfig({ ...cfg, initialized: true });
            setIsAdmin(true);
            setView('admin');
            showToast('Concursul a fost inițializat! 🎉');
          }} showToast={showToast} />
          <Toast toast={toast} onClose={() => setToast(null)} />
        </>;
      }

      return (
        <div className="min-h-screen text-stone-100" style={{ background: `radial-gradient(ellipse 80% 60% at 50% 0%, rgba(220, 38, 38, 0.15), transparent 60%), radial-gradient(ellipse 60% 80% at 100% 100%, rgba(37, 99, 235, 0.1), transparent 60%), linear-gradient(180deg, #0c0a09 0%, #0c0a09 100%)` }}>
          <Header view={view} setView={setView} currentPlayer={currentPlayer} setCurrentPlayer={setCurrentPlayer} isAdmin={isAdmin} setIsAdmin={setIsAdmin} isLocked={isLocked} savingState={savingState} onRefresh={loadState} />
          <main className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 py-6 sm:py-10 relative">
            {view === 'home' && <HomeView config={config} isLocked={isLocked} players={players} standings={standings} setView={setView} currentPlayer={currentPlayer} setCurrentPlayer={setCurrentPlayer} savePlayers={savePlayers} isAdmin={isAdmin} setIsAdmin={setIsAdmin} adminPassword={config.adminPassword} showToast={showToast} />}
            {view === 'predict' && currentPlayer && <PredictView player={currentPlayer} isLocked={isLocked} results={results} scoring={config.scoring} updatePlayer={(updated) => { const newPlayers = players.map(p => p.name === updated.name ? updated : p); setCurrentPlayer(updated); savePlayers(newPlayers); }} />}
            {view === 'leaderboard' && <LeaderboardView standings={standings} players={players} results={results} scoring={config.scoring} />}
            {view === 'admin' && isAdmin && <AdminView config={config} results={results} saveConfig={saveConfig} saveResults={saveResults} players={players} savePlayers={savePlayers} showToast={showToast} askConfirm={askConfirm} />}
          </main>
          <footer className="border-t border-stone-800 mt-16 py-6 text-center text-xs text-stone-500 font-mono tracking-widest">
            <span style={{fontFamily:'"Bebas Neue", sans-serif', fontSize: 16, letterSpacing: '0.1em'}}>WC26 PREDICTOR</span>
            <span className="mx-3 text-stone-700">·</span>
            72 MECIURI · GRUPELE A-L · 11–28 IUNIE 2026
          </footer>
          <Toast toast={toast} onClose={() => setToast(null)} />
          <ConfirmDialog open={!!confirmDlg} title={confirmDlg?.title} message={confirmDlg?.message} danger={confirmDlg?.danger}
            onConfirm={() => { confirmDlg?.onConfirm(); setConfirmDlg(null); }} onCancel={() => setConfirmDlg(null)} />
        </div>
      );
    }

    function ScoringInput({ label, value, onChange, hint, color }) {
      const colors = { amber: { border: '#f59e0b', text: '#fcd34d' }, emerald: { border: '#10b981', text: '#6ee7b7' }, sky: { border: '#0ea5e9', text: '#7dd3fc' } };

    function PredictView({ player, isLocked, results, scoring, updatePlayer }) {
      const [filter, setFilter] = useState('all');
      const [search, setSearch] = useState('');
      const filtered = MATCHES.filter(m => {
        if (filter !== 'all' && m.g !== filter) return false;
        if (search && !`${m.e1} ${m.e2}`.toLowerCase().includes(search.toLowerCase())) return false;
        return true;
      });
      const completed = MATCHES.filter(m => { const p = player.predictions?.[m.n]; return p && p.g1 !== '' && p.g2 !== '' && p.g1 != null && p.g2 != null; }).length;
      const updatePred = (matchN, field, value) => {
        if (isLocked) return;
        const v = value === '' ? '' : Math.max(0, Math.min(15, parseInt(value) || 0));
        const newPreds = { ...player.predictions, [matchN]: { ...(player.predictions?.[matchN] || {}), [field]: v } };
        updatePlayer({ ...player, predictions: newPreds });
      };
      return (
        <div className="space-y-6">
          <div className="flex flex-col sm:flex-row sm:items-end justify-between gap-4">
            <div>
              <div className="text-xs tracking-[0.25em] text-amber-400 font-mono">PRONOSTICURI · {player.name.toUpperCase()}</div>
              <h2 className="text-3xl sm:text-4xl font-black mt-1" style={{fontFamily:'"Bebas Neue", sans-serif', letterSpacing:'0.02em'}}>COMPLETEAZĂ-ȚI <span className="text-amber-400">SCORURILE</span></h2>

            </div>
            {!isLocked && (

                <div className="flex justify-between text-[10px] tracking-widest font-mono text-stone-500 mb-1"><span>PROGRES</span><span>{Math.round(completed/72*100)}%</span></div>

              </div>

'bg-amber-400 text-stone-950' : 'bg-stone-800 text-stone-400'}`} style={{fontFamily:'"Bebas Neue", sans-serif'}}>TOATE</button>

'text-stone-950' : 'text-stone-400'}`} style={{fontFamily:'"Bebas Neue", sans-serif', backgroundColor: filter===g ? groupColors[g].border : 'transparent', border: `1px solid ${groupColors[g].border}`}}>GR. {g}</button>

    function MatchCard({ match, pred, real, scoring, isLocked, onUpdate }) {

      const hasResult = real && real.g1 !== '' && real.g2 !== '' && real.g1 != null && real.g2 != null;
      let pointBadge = null;
      if (points != null) {
        let bg = 'bg-stone-700 text-stone-300';
        if (points === scoring.exact) bg = 'bg-amber-400 text-stone-950';
        else if (points === scoring.diff) bg = 'bg-emerald-500 text-stone-950';
        else if (points === scoring.outcome) bg = 'bg-sky-500 text-white';
        pointBadge = <div className={`${bg} px-2 py-1 rounded text-xs font-black`} style={{fontFamily:'"Bebas Neue", sans-serif'}}>+{points} PCT</div>;
      }
      const inputBg = isLocked ? '#1c1917' : '#fef3c7';
      const inputColor = isLocked ? '#fef3c7' : '#0c0a09';
      const inputBorder = `2px solid ${isLocked ? '#44403c' : '#fbbf24'}`;
      return (
        <div className="rounded-xl border bg-stone-900/60 backdrop-blur p-4 transition" style={{borderColor: c.border + '60'}}>
          <div className="flex items-center justify-between mb-3">
            <div className="flex items-center gap-2">
              <span className="px-2 py-0.5 text-[10px] font-black rounded font-mono tracking-widest" style={{backgroundColor: c.bg, color: c.text, border: `1px solid ${c.border}40`}}>GR. {m.g}</span>
              <span className="text-[10px] text-stone-500 font-mono tracking-widest">M{String(m.n).padStart(2,'0')}</span>
            </div>
            <div className="text-[10px] text-stone-400 font-mono tracking-wider">{m.d} · {m.t}</div>
          </div>
          <div className="space-y-2">
            <div className="flex items-center gap-3">
              <span className="text-2xl shrink-0 w-8 text-center">{m.f1}</span>
              <span className="font-bold text-sm flex-1 min-w-0 truncate">{m.e1}</span>
              <input type="number" inputMode="numeric" min="0" max="15" value={pred?.g1 ?? ''} onChange={e => onUpdate(m.n, 'g1', e.target.value)} disabled={isLocked} placeholder="-" className="w-14 h-12 text-center rounded-lg font-black text-xl outline-none transition disabled:opacity-60 shrink-0" style={{ fontFamily:'"Bebas Neue", sans-serif', backgroundColor: inputBg, color: inputColor, border: inputBorder, WebkitAppearance:'none' }} />

              <span className="text-2xl shrink-0 w-8 text-center">{m.f2}</span>

''} onChange={e => onUpdate(m.n, 'g2', e.target.value)} disabled={isLocked} placeholder="-" className="w-14 h-12 text-center rounded-lg font-black text-xl outline-none transition disabled:opacity-60 shrink-0" style={{ fontFamily:'"Bebas Neue", sans-serif', backgroundColor: inputBg, color: inputColor, border: inputBorder, WebkitAppearance:'none' }} />
            </div>
          </div>
          {hasResult && (
            <div className="mt-3 pt-3 border-t border-stone-800 flex items-center justify-between text-xs">
              <div className="flex items-center gap-2"><div className="text-stone-500 font-mono tracking-widest text-[10px]">REZULTAT REAL</div><div className="font-black text-emerald-400" style={{fontFamily:'"Bebas Neue", sans-serif', fontSize: 18}}>{real.g1} – {real.g2}</div></div>
              {pointBadge}

          {standings.length === 0 ? <div className="bg-stone-900/60 border border-stone-800 rounded-xl p-12 text-center text-stone-500">Niciun jucător încă. Înscrieți-vă din pagina principală!</div>
          : <div className="bg-stone-900/60 backdrop-blur border border-stone-800 rounded-xl overflow-hidden">
              <div className="hidden sm:grid grid-cols-[60px,1fr,80px,80px,80px,80px,100px] gap-2 px-4 py-3 border-b border-stone-800 bg-stone-950/50 text-[10px] tracking-widest text-stone-500 font-mono">
                <div>LOC</div><div>JUCĂTOR</div><div className="text-center">EXACTE</div><div className="text-center">DIFER.</div><div className="text-center">1X2</div><div className="text-center">GREȘ.</div><div className="text-right">TOTAL</div>
              </div>
              {standings.map((s, i) => {
                const player = players.find(p => p.name === s.name);
                return (
                  <div key={s.name}>
                    <button type="button" onClick={() => setShowDetail(showDetail === s.name ? null : s.name)} className={`w-full text-left px-4 py-4 border-b border-stone-800 last:border-0 hover:bg-stone-800/30 transition ${i === 0 && s.played > 0 ? 'bg-amber-400/5' : ''}`}>
                      <div className="hidden sm:grid grid-cols-[60px,1fr,80px,80px,80px,80px,100px] gap-2 items-center">
                        <div><div className={`w-9 h-9 flex items-center justify-center rounded-md font-black ${i === 0 ? 'bg-amber-400 text-stone-950' : i === 1 ? 'bg-stone-300 text-stone-950' : i === 2 ? 'bg-orange-700 text-white' : 'bg-stone-800 text-stone-400'}`} style={{fontFamily:'"Bebas Neue", sans-serif', fontSize: 18}}>{i + 1}</div></div>
                        <div className="flex items-center gap-2 min-w-0"><div className="font-bold truncate">{s.name}</div>{i === 0 && s.played > 0 && <span className="text-lg">👑</span>}</div>
                        <div className="text-center"><span className="text-amber-400 font-bold">{s.exact}</span></div>
                        <div className="text-center"><span className="text-emerald-400 font-bold">{s.diff}</span></div>
                        <div className="text-center"><span className="text-sky-400 font-bold">{s.outcome}</span></div>
                        <div className="text-center"><span className="text-stone-500 font-bold">{s.wrong}</span></div>
                        <div className="text-right flex items-center justify-end gap-2"><div className="text-3xl font-black text-amber-400 leading-none" style={{fontFamily:'"Bebas Neue", sans-serif'}}>{s.total}</div><IconChevronDown size={14} className={`text-stone-500 transition ${showDetail === s.name ? 'rotate-180' : ''}`} /></div>
                      </div>
                      <div className="sm:hidden flex items-center gap-3">
                        <div className={`w-10 h-10 flex items-center justify-center rounded-md font-black shrink-0 ${i === 0 ? 'bg-amber-400 text-stone-950' : i === 1 ? 'bg-stone-300 text-stone-950' : i === 2 ? 'bg-orange-700 text-white' : 'bg-stone-800 text-stone-400'}`} style={{fontFamily:'"Bebas Neue", sans-serif', fontSize: 18}}>{i + 1}</div>
                        <div className="flex-1 min-w-0">
                          <div className="flex items-center gap-1.5"><div className="font-bold truncate">{s.name}</div>{i === 0 && s.played > 0 && <span>👑</span>}</div>
                          <div className="text-[10px] text-stone-500 font-mono tracking-widest mt-0.5"><span className="text-amber-400">{s.exact}E</span><span className="mx-1.5">·</span><span className="text-emerald-400">{s.diff}D</span><span className="mx-1.5">·</span><span className="text-sky-400">{s.outcome}R</span></div>
                        </div>
                        <div className="flex items-center gap-2 shrink-0">
                          <div className="text-right"><div className="text-3xl font-black text-amber-400 leading-none" style={{fontFamily:'"Bebas Neue", sans-serif'}}>{s.total}</div><div className="text-[9px] text-stone-500 font-mono tracking-widest mt-0.5">PCT</div></div>
                          <IconChevronDown size={14} className={`text-stone-500 transition ${showDetail === s.name ? 'rotate-180' : ''}`} />
                        </div>
                      </div>
                    </button>
                    {showDetail === s.name && player && <PlayerDetail player={player} results={results} scoring={scoring} />}
                  </div>
                );
              })}
            </div>}
        </div>
      );
    }

    function PlayerDetail({ player, results, scoring }) {
      return (
        <div className="bg-stone-950/50 border-b border-stone-800 px-4 py-4">
          <div className="text-xs tracking-widest font-mono text-stone-500 mb-3">PRONOSTICURILE LUI {player.name.toUpperCase()}</div>
          <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-2">
            {MATCHES.map(m => {
              const pred = player.predictions?.[m.n];
              const real = results[m.n];
              const pts = calcPoints(pred, real, scoring);
              const c = groupColors[m.g];
              let badge = '';
              if (pts === scoring.exact) badge = 'bg-amber-400 text-stone-950';
              else if (pts === scoring.diff) badge = 'bg-emerald-500 text-stone-950';
              else if (pts === scoring.outcome) badge = 'bg-sky-500 text-white';
              else if (pts === scoring.wrong) badge = 'bg-stone-700 text-stone-400';
              return (
                <div key={m.n} className="bg-stone-900 border rounded-md px-2.5 py-1.5 text-xs flex items-center justify-between gap-2" style={{borderColor: c.border + '40'}}>
                  <div className="flex items-center gap-2 min-w-0">
                    <span className="text-[9px] font-mono text-stone-500">M{m.n}</span>
                    <span className="truncate text-stone-300 text-[11px]">{m.f1}{m.e1.slice(0,3)} – {m.e2.slice(0,3)}{m.f2}</span>
                  </div>
                  <div className="flex items-center gap-1.5 shrink-0">
                    <span className="font-mono font-bold text-stone-200">{pred?.g1 ?? '–'}-{pred?.g2 ?? '–'}</span>
                    {real && real.g1 != null && real.g2 != null && <span className="font-mono text-emerald-400">[{real.g1}-{real.g2}]</span>}
                    {pts != null && <span className={`${badge} px-1.5 rounded text-[10px] font-black`}>+{pts}</span>}
                  </div>
                </div>
              );
            })}
          </div>
        </div>
      );
    }

    function AdminView({ config, results, saveConfig, saveResults, players, savePlayers, showToast, askConfirm }) {
      const [tab, setTab] = useState('results');
      return (
        <div className="space-y-6">
          <div>
            <div className="text-xs tracking-[0.25em] text-red-400 font-mono">🔐 PANOU ADMINISTRARE</div>
            <h2 className="text-3xl sm:text-4xl font-black mt-1" style={{fontFamily:'"Bebas Neue", sans-serif', letterSpacing:'0.02em'}}>CONTROL <span className="text-red-400">TOTAL</span></h2>
          </div>
          <div className="flex gap-1 border-b border-stone-800 overflow-x-auto -mx-4 px-4 sm:mx-0 sm:px-0 no-scrollbar">
            {[['results', '🥅 REZULTATE'], ['settings', '⚙️ SETĂRI'], ['players', '👥 JUCĂTORI']].map(([id, label]) => (
              <button type="button" key={id} onClick={() => setTab(id)} className={`px-4 py-3 text-xs font-bold tracking-widest transition whitespace-nowrap ${tab===id ? 'border-b-2 border-amber-400 text-amber-400' : 'text-stone-500 hover:text-stone-300'}`} style={{fontFamily:'"Bebas Neue", sans-serif', letterSpacing:'0.15em'}}>{label}</button>
            ))}
          </div>
          {tab === 'results' && <AdminResults results={results} saveResults={saveResults} showToast={showToast} />}
          {tab === 'settings' && <AdminSettings config={config} saveConfig={saveConfig} showToast={showToast} />}
          {tab === 'players' && <AdminPlayers players={players} savePlayers={savePlayers} askConfirm={askConfirm} showToast={showToast} />}
        </div>
      );
    }

    function AdminResults({ results, saveResults, showToast }) {
      const [filter, setFilter] = useState('all');
      const [pendingResults, setPendingResults] = useState(results);
      const [dirty, setDirty] = useState(false);
      useEffect(() => { setPendingResults(results); setDirty(false); }, [results]);
      const update = (n, field, value) => {
        const v = value === '' ? '' : Math.max(0, Math.min(15, parseInt(value) || 0));
        setPendingResults(prev => ({ ...prev, [n]: { ...(prev[n] || {}), [field]: v } }));
        setDirty(true);
      };
      const save = async () => {
        const clean = {};
        for (const [k, v] of Object.entries(pendingResults)) {
          if (v && v.g1 !== '' && v.g2 !== '' && v.g1 != null && v.g2 != null) clean[k] = { g1: +v.g1, g2: +v.g2 };
        }
        await saveResults(clean);
        setDirty(false);
        showToast('Rezultate salvate. Punctajele s-au actualizat.');
      };
      const filtered = MATCHES.filter(m => filter === 'all' || m.g === filter);
      const completed = MATCHES.filter(m => pendingResults[m.n] && pendingResults[m.n].g1 !== '' && pendingResults[m.n].g2 !== '' && pendingResults[m.n].g1 != null && pendingResults[m.n].g2 != null).length;
      return (
        <div className="space-y-4">
          <div className="bg-emerald-500/10 border border-emerald-500/30 rounded-lg p-4 flex flex-col sm:flex-row gap-3 items-start sm:items-center justify-between">
            <div><div className="font-bold text-emerald-300">📊 {completed}/72 meciuri completate</div><div className="text-xs text-stone-400 mt-1">Punctajele jucătorilor se actualizează automat după salvare.</div></div>
            {dirty && <button type="button" onClick={save} className="bg-amber-400 text-stone-950 px-5 py-2.5 rounded-lg font-black hover:bg-amber-300 transition flex items-center gap-2 shrink-0" style={{fontFamily:'"Bebas Neue", sans-serif', letterSpacing:'0.15em'}}><IconSave size={14}/> SALVEAZĂ</button>}
          </div>
          <div className="flex gap-2 overflow-x-auto -mx-4 px-4 sm:mx-0 sm:px-0 sm:flex-wrap pb-1 no-scrollbar">
            <button type="button" onClick={() => setFilter('all')} className={`px-3 py-1.5 rounded-md text-xs font-bold tracking-widest whitespace-nowrap shrink-0 ${filter==='all' ? 'bg-amber-400 text-stone-950' : 'bg-stone-800 text-stone-400'}`} style={{fontFamily:'"Bebas Neue", sans-serif'}}>TOATE</button>
            {['A','B','C','D','E','F','G','H','I','J','K','L'].map(g => (
              <button type="button" key={g} onClick={() => setFilter(g)} className={`px-3 py-1.5 rounded-md text-xs font-bold tracking-widest whitespace-nowrap shrink-0 ${filter===g ? 'text-stone-950' : 'text-stone-400'}`} style={{fontFamily:'"Bebas Neue", sans-serif', backgroundColor: filter===g ? groupColors[g].border : 'transparent', border: `1px solid ${groupColors[g].border}`}}>GR. {g}</button>
            ))}
          </div>
          <div className="grid grid-cols-1 md:grid-cols-2 gap-3">
            {filtered.map(m => {
              const r = pendingResults[m.n];
              const c = groupColors[m.g];
              return (
                <div key={m.n} className="rounded-xl border bg-stone-900/60 p-4" style={{borderColor: c.border + '40'}}>
                  <div className="flex items-center justify-between mb-3">
                    <span className="px-2 py-0.5 text-[10px] font-black rounded font-mono tracking-widest" style={{backgroundColor: c.bg, color: c.text}}>GR. {m.g} · M{m.n}</span>
                    <span className="text-[10px] text-stone-500 font-mono">{m.d} · {m.t}</span>
                  </div>
                  <div className="space-y-2">
                    <div className="flex items-center gap-3">
                      <span className="text-2xl shrink-0 w-8 text-center">{m.f1}</span>
                      <span className="font-bold text-sm flex-1 min-w-0 truncate">{m.e1}</span>
                      <input type="number" inputMode="numeric" min="0" max="15" value={r?.g1 ?? ''} onChange={e => update(m.n, 'g1', e.target.value)} placeholder="–" className="w-14 h-12 text-center rounded-lg font-black text-xl outline-none shrink-0" style={{fontFamily:'"Bebas Neue", sans-serif', backgroundColor:'#dcfce7', color:'#0c0a09', border:'2px solid #22c55e', WebkitAppearance:'none'}} />
                    </div>
                    <div className="flex items-center gap-3">
                      <span className="text-2xl shrink-0 w-8 text-center">{m.f2}</span>
                      <span className="font-bold text-sm flex-1 min-w-0 truncate">{m.e2}</span>
                      <input type="number" inputMode="numeric" min="0" max="15" value={r?.g2 ?? ''} onChange={e => update(m.n, 'g2', e.target.value)} placeholder="–" className="w-14 h-12 text-center rounded-lg font-black text-xl outline-none shrink-0" style={{fontFamily:'"Bebas Neue", sans-serif', backgroundColor:'#dcfce7', color:'#0c0a09', border:'2px solid #22c55e', WebkitAppearance:'none'}} />
                    </div>
                  </div>
                </div>
              );
            })}
          </div>
          {dirty && <div className="sticky bottom-4 flex justify-end">
            <button type="button" onClick={save} className="bg-amber-400 text-stone-950 px-6 py-3 rounded-lg font-black hover:bg-amber-300 transition flex items-center gap-2 shadow-2xl shadow-amber-500/30" style={{fontFamily:'"Bebas Neue", sans-serif', letterSpacing:'0.2em'}}><IconSave size={16}/> SALVEAZĂ MODIFICĂRILE</button>
          </div>}
        </div>
      );
    }

    function AdminSettings({ config, saveConfig, showToast }) {
      const [scoring, setScoring] = useState(config.scoring);
      const [deadline, setDeadline] = useState(config.deadline);
      const [pwd, setPwd] = useState(config.adminPassword);
      const save = async () => {
        await saveConfig({ ...config, scoring: { ...scoring, wrong: 0 }, deadline, adminPassword: pwd });
        showToast('Setări salvate cu succes!');
      };
      return (
        <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
          <div className="bg-stone-900/60 border border-stone-800 rounded-xl p-6 space-y-5">
            <h3 className="font-black tracking-widest text-lg" style={{fontFamily:'"Bebas Neue", sans-serif', letterSpacing:'0.15em'}}>PUNCTAJE</h3>
            <div className="space-y-3">
              <SettingsRow label="Scor exact" desc="2-1 → 2-1" value={scoring.exact} onChange={v => setScoring({...scoring, exact: +v || 0})} accent="amber" />
              <SettingsRow label="Diferență de gol" desc="2-1 → 3-2" value={scoring.diff} onChange={v => setScoring({...scoring, diff: +v || 0})} accent="emerald" />
              <SettingsRow label="Rezultat 1/X/2" desc="2-1 → 4-0" value={scoring.outcome} onChange={v => setScoring({...scoring, outcome: +v || 0})} accent="sky" />
            </div>
          </div>
          <div className="bg-stone-900/60 border border-stone-800 rounded-xl p-6 space-y-5">
            <h3 className="font-black tracking-widest text-lg" style={{fontFamily:'"Bebas Neue", sans-serif', letterSpacing:'0.15em'}}>DEADLINE & SECURITATE</h3>
            <div><Label>DEADLINE PRONOSTICURI</Label><input type="datetime-local" value={deadline} onChange={e => setDeadline(e.target.value)} style={inputStyleObj} /><div className="text-xs text-stone-500 mt-1">După această oră jucătorii nu mai pot edita.</div></div>
            <div><Label>PAROLĂ ADMIN</Label><input type="password" value={pwd} onChange={e => setPwd(e.target.value)} style={inputStyleObj} autoComplete="new-password" /></div>
          </div>
          <button type="button" onClick={save} className="lg:col-span-2 bg-amber-400 text-stone-950 px-6 py-3 rounded-lg font-black hover:bg-amber-300 transition flex items-center justify-center gap-2" style={{fontFamily:'"Bebas Neue", sans-serif', letterSpacing:'0.2em'}}><IconSave size={16}/> SALVEAZĂ SETĂRILE</button>
        </div>
      );
    }

    function SettingsRow({ label, desc, value, onChange, accent }) {
      const colors = { amber: '#fbbf24', emerald: '#10b981', sky: '#0ea5e9' };
      return (
        <div className="flex items-center justify-between gap-3">
          <div><div className="font-bold text-sm">{label}</div><div className="text-[11px] text-stone-500 font-mono">{desc}</div></div>
          <input type="number" inputMode="numeric" min="0" max="20" value={value} onChange={e => onChange(e.target.value)} className="w-20 text-center rounded-md py-2 font-black text-xl bg-stone-950 border-2 outline-none" style={{fontFamily:'"Bebas Neue", sans-serif', borderColor: colors[accent], color: colors[accent], WebkitAppearance:'none'}} />
        </div>
      );
    }

    function AdminPlayers({ players, savePlayers, askConfirm, showToast }) {
      const remove = (name) => {
        askConfirm('ȘTERGE JUCĂTOR', `Sigur vrei să ștergi jucătorul "${name}"? Toate pronosticurile lui vor fi pierdute.`,
          async () => { await savePlayers(players.filter(p => p.name !== name)); showToast(`Jucătorul "${name}" a fost șters.`); }, true);
      };
      return (
        <div className="bg-stone-900/60 border border-stone-800 rounded-xl overflow-hidden">
          <div className="px-4 py-3 border-b border-stone-800 bg-stone-950/50 text-xs tracking-widest text-stone-500 font-mono">{players.length} JUCĂTORI ÎNSCRIȘI</div>
          {players.length === 0 ? <div className="p-8 text-center text-stone-500">Niciun jucător încă.</div>
          : players.map(p => {
            const completed = MATCHES.filter(m => p.predictions?.[m.n] && p.predictions[m.n].g1 !== '' && p.predictions[m.n].g2 !== '' && p.predictions[m.n].g1 != null && p.predictions[m.n].g2 != null).length;
            return (
              <div key={p.name} className="px-4 py-3 border-b border-stone-800 last:border-0 flex items-center justify-between gap-4">
                <div><div className="font-bold">{p.name}</div><div className="text-xs text-stone-500 font-mono">{completed}/72 pronosticuri · parolă: {'•'.repeat(p.password.length)}</div></div>
                <button type="button" onClick={() => remove(p.name)} className="text-red-400 hover:text-red-300 p-2 rounded hover:bg-red-500/10" title="Șterge"><IconTrash size={16}/></button>
              </div>
            );
          })}
        </div>
      );
    }
